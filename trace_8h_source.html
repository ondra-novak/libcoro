173  auto await_suspend(std::coroutine_handle<> h, std::source_location loc = std::source_location::current()) {
174  using RetVal = decltype(_awt.await_suspend(h));
175  if constexpr(!std::is_same_v<std::decay_t<X>, ident_awt>) {
176  impl::_instance.on_await_on(h.address(), &_awt, typeid(X).name());

179  std::coroutine_handle<> r = proxy_await_suspend(_awt,h, loc);
180  impl::_instance.on_switch(h.address(), r.address(), &loc);
181  return r;
