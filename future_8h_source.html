982  template<std::invocable<> Awt, typename ResumeFn >
983  RegAwtRes register_awaiter(Awt &&awt, ResumeFn &&resumeFn) {
984  //lock awaiter slot
985  State st = _state.exchange(State::evaluating);
986  switch (st) {
987  //already resolved?
988  case State::resolved: _state.store(st); //restore state
989  return RegAwtRes::resolved;
990 
991  case State::pending: //<still pending? move awaiter
992  new(&_awaiter) auto(to_awaiter(std::forward<Awt>(awt)));
993  break;
994  default:
995  case State::evaluating: //<locked - this is error
996  throw still_pending_exception();
997  case State::awaited: //<already awaited? destroy previous awaiter and construct new one
998  std::destroy_at(&_awaiter);
999  new(&_awaiter) auto(to_awaiter(std::forward<Awt>(awt)));
1000  break;
1001  case State::deferred: //<deferred ? perform evaluation and try again
1002  if (!startDeferredEvaluation(resumeFn)) return RegAwtRes::resolved;
1003  return register_awaiter(std::forward<Awt>(awt),std::forward<ResumeFn>(resumeFn));
1004  }
1005  st = State::evaluating; //replace evaluating with awaited
1006  if (!_state.compare_exchange_strong(st, State::awaited)) {
1007  //if failed, it should be resolved
1008  //mark that awaited must be clean up (always)
1009  _awaiter_cleanup = true;
1010  return RegAwtRes::constructed_resolved;
1011  }
1012  //awaiter has been set
1013  return RegAwtRes::constructed_ready;
1014  }
