1045  template<std::invocable<> Awt, typename ResumeFn >
1046  RegAwtRes register_awaiter(Awt &&awt, ResumeFn &&resumeFn) {
1047  //lock awaiter slot
1048  State st = _state.exchange(State::evaluating);
1049  switch (st) {
1050  //already resolved?
1051  case State::resolved: _state.store(st); //restore state
1052  return RegAwtRes::resolved;
1053 
1054  case State::pending: //<still pending? move awaiter
1055  new(&_awaiter) auto(to_awaiter(std::forward<Awt>(awt)));
1056  break;
1057  default:
1058  case State::evaluating: //<locked - this is error
1059  throw still_pending_exception();
1060  case State::awaited: //<already awaited? destroy previous awaiter and construct new one
1061  std::destroy_at(&_awaiter);
1062  new(&_awaiter) auto(to_awaiter(std::forward<Awt>(awt)));
1063  break;
1064  case State::deferred: //<deferred ? perform evaluation and try again
1065  if (!startDeferredEvaluation(resumeFn)) return RegAwtRes::resolved;
