938  template<std::invocable<> Awt, typename ResumeFn >
939  RegAwtRes register_awaiter(Awt &&awt, ResumeFn &&resumeFn) {
940 
941  //determine, whether function returns coroutine_handle
942  if constexpr(std::is_invocable_r_v<std::coroutine_handle<>, Awt>) {
943  //lock awaiter slot
944  State st = _state.exchange(State::evaluating);
945  switch (st) {
946  //already resolved?
947  case State::resolved: _state.store(st); //restore state
948  return RegAwtRes::resolved;
949 
950  case State::pending: //<still pending? move awaiter
951  std::construct_at(&_awaiter, std::forward<Awt>(awt));
952  break;
953  default:
954  case State::evaluating: //<locked - this is error
955  throw still_pending_exception();
956  case State::awaited: //<already awaited? destroy previous awaiter and construct new one
957  std::destroy_at(&_awaiter);
958  std::construct_at(&_awaiter, std::forward<Awt>(awt));
959  break;
960  case State::deferred: //<deferred ? perform evaluation and try again
961  if (!startDeferredEvaluation(resumeFn)) return RegAwtRes::resolved;
962  return register_awaiter(std::forward<Awt>(awt),std::forward<ResumeFn>(resumeFn));
963  }
964  st = State::evaluating; //replace evaluating with awaited
965  if (!_state.compare_exchange_strong(st, State::awaited)) {
966  //if failed, it should be resolved
967  //mark that awaited must be clean up (always)
968  _awaiter_cleanup = true;
969  return RegAwtRes::constructed_resolved;
970  }
971  //awaiter has been set
972  return RegAwtRes::constructed_ready;
973  } else {
974  //if not, wrap the function into function returning empty coroutine handle
975  return register_awaiter([awt = std::move(awt)]() mutable {
976  awt(); return std::coroutine_handle<>();
977  }, std::forward<ResumeFn>(resumeFn));
978  }
