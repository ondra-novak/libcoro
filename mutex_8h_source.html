277  //idx out of range? success
278  if constexpr(idx >= std::tuple_size_v<ownership>) {
279  return true;
280  } else {

282  if (idx == skip) {

284  } else {

287  //retrieve lock
288  auto &mx = std::get<idx>(_mxlist);
289  //try to lock
290  own = std::get<idx>(_mxlist).try_lock();
291  //we success
292  if (own) {
293  //continue by next lock
