29 generator<T, Alloc> aggregator(Alloc &, std::vector<generator<T> > gens) {
30 
31  //list of futures waiting for results from generators
32  std::vector<deferred_future<T> > futures;
33  //queue - stores index to resolved future
34  queue<std::size_t> q;
35  std::size_t cnt = gens.size();
36 
37  //activate function - activate generator and awaits it passing index to queue
38  auto activate = [&](std::size_t idx) {
39  futures[idx] = gens[idx]();
40  futures[idx] >> [&q, idx]{q.push(idx);};
41  };
42 
43  //prepare array of futures
44  futures.resize(cnt);
45 
46  //activate all generators
47  for (std::size_t i = 0; i < cnt; ++i) activate(i);
48 
49  //define function, which is called on exit (including destroy)
50  on_leave lv=[&]{
51 
53  bool pending = false;
54  for (auto &f: futures) {
55  //this should return false, as the future is already resolved
56  //but when returns true, we need handle this as special case
57  //disarming all futures allows to detach it from the queue and detect such state
58  auto r = f.set_callback([]{});
59  pending = pending || r;
60  }
61  //if such future exists, install a coroutine, which awaits to resolution
62  if (pending) {
63  //coroutine
64  auto dtch = [](std::vector<generator<T> > , std::vector<deferred_future<T> > futures) -> coroutine {
65  //cycle over all futures and @b co_await for just has_value - we don't need the value
66  for (auto &f: futures) {
67  co_await f.wait();
68  }
69 
70  };
71  //move futures and generators to the coroutine
72  dtch(std::move(gens), std::move(futures));
73  }
74  };
75 
76  //any running generator?
77  while (cnt) {
78 
79  //wait on queue
80  std::size_t idx = co_await q.pop();
81  //retrieve index, if it has value
82  if (futures[idx].has_value()) {
83  //yield value of future
84  co_yield std::move(futures[idx]).get();
85  //activate the generator
86  activate(idx);
87  } else {
88  //if generator has no value, decrease count of running generators
89  --cnt;
90  }
91  }
